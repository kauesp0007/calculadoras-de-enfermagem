    <script>
      // Função auxiliar para carregar HTML modular
      function fetchHTML(url, containerId) {
        fetch(url)
          .then((response) => {
            if (!response.ok) {
              throw new Error(`Não foi possível carregar ${url}`);
            }
            return response.text();
          })
          .then((data) => {
            const container = document.getElementById(containerId);
            if (container) {
              container.innerHTML = data;
            } else {
              console.warn(`Container com ID '${containerId}' não encontrado.`);
            }
          })
          .catch((error) => console.error(`Erro ao buscar ${url}:`, error));
      }

      // Carrega os módulos HTML
      document.addEventListener("DOMContentLoaded", () => {
        // O menu-global.html é pré-carregado, mas o 'global-scripts.js'
        // provavelmente já lida com o carregamento dele no 'global-header-container'.
        // Se não, descomente a linha abaixo:
        // fetchHTML("/menu-global.html", "global-header-container");

        fetchHTML("/_language_selector.html", "language-dropdown-wrapper");
        fetchHTML("/footer.html", "footer-placeholder");
        fetchHTML(
          "/global-body-elements.html",
          "global-body-elements-container"
        );
      });
    </script>
