    document.addEventListener('DOMContentLoaded', () => {

          // Remove o foco do input para garantir que o scroll continue na página
          this.blur();
        });
      });
      const resultadoDiv = document.getElementById('resultado');
      const btnCalcular = document.getElementById('btnCalcular');
      const btnLimpar = document.getElementById('btnLimpar');
      const btnGerarPDF = document.getElementById('btnGerarPDF');
      const btnNandaNicNoc = document.getElementById('btnNandaNicNoc');
      const btnGrauComplexidade = document.getElementById('btnGrauComplexidade');
      const tipoUnidadeSelect = document.getElementById('tipoUnidade');
      const pacientesUnidadeInternacaoDiv = document.getElementById('pacientesUnidadeInternacao');

      // Mapeamento dos HPPD e percentuais de acordo com a Resolução COFEN Nº 543/2017
      const HPPD = {
        'cuidadosMinimos': 3.8,
        'cuidadosIntermediarios': 5.6,
        'altaDependencia': 9.4,
        'semiIntensivos': 10.0,
        'intensivos': 17.9
      };

      const percentuais = {
        'cuidadosMinimos': { enfermeiro: 0.33, tecnicoAuxiliar: 0.67 },
        'cuidadosIntermediarios': { enfermeiro: 0.33, tecnicoAuxiliar: 0.67 },
        'altaDependencia': { enfermeiro: 0.36, tecnicoAuxiliar: 0.64 },
        'semiIntensivos': { enfermeiro: 0.42, tecnicoAuxiliar: 0.58 },
        'intensivos': { enfermeiro: 0.52, tecnicoAuxiliar: 0.48 }
      };

      const cargaHorariaSemanal = {
        '30': 30,
        '36': 36,
        '40': 40,
        '44': 44
      };

      function toggleUnidadeInputs() {
        if (tipoUnidadeSelect.value === 'unidadeInternacao') {
          pacientesUnidadeInternacaoDiv.classList.remove('hidden');
          pacientesUTIDiv.classList.add('hidden');
        } else {
          pacientesUnidadeInternacaoDiv.classList.add('hidden');
          pacientesUTIDiv.classList.remove('hidden');
        }

      function calcularDimensionamento() {
        const cargaSelecionada = document.getElementById('cargaHoraria').value;
        const tipoUnidade = document.getElementById('tipoUnidade').value;

        let totalHorasEnfermagemDia = 0;
        let maiorCargaTrabalhoTipo = '';

        let hasError = false;

        const numCM = parseInt(document.getElementById('numCM').value, 10) || 0;
        const numCI = parseInt(document.getElementById('numCI').value, 10) || 0;
        const numCAD = parseInt(document.getElementById('numCAD').value, 10) || 0;
        const numCSI = parseInt(document.getElementById('numCSI').value, 10) || 0;
        const numLeitosUTI = parseInt(document.getElementById('numLeitosUTI').value, 10) || 0;

        // Limpa mensagens de erro
        document.getElementById('cargaHoraria-error').classList.add('hidden');
        document.getElementById('tipoUnidade-error').classList.add('hidden');
        document.getElementById('numCM-error').classList.add('hidden');
        document.getElementById('numCI-error').classList.add('hidden');
        document.getElementById('numCAD-error').classList.add('hidden');
        document.getElementById('numCSI-error').classList.add('hidden');
        document.getElementById('numLeitosUTI-error').classList.add('hidden');

        if (tipoUnidade === 'uti') {
          if (numLeitosUTI === 0) {
            document.getElementById('numLeitosUTI-error').textContent = 'Por favor, insira um número de leitos de UTI maior que 0.';
            document.getElementById('numLeitosUTI-error').classList.remove('hidden');
            hasError = true;
          }
          totalHorasEnfermagemDia = numLeitosUTI * HPPD['intensivos'];
          maiorCargaTrabalhoTipo = 'intensivos';
        } else { // Unidade de Internação
          const totalPacientes = numCM + numCI + numCAD + numCSI;
          if (totalPacientes === 0) {
            document.getElementById('numCM-error').textContent = 'Por favor, insira o número de pacientes para pelo menos um tipo de cuidado.';
            document.getElementById('numCM-error').classList.remove('hidden');
            hasError = true;
          }

          const horasCM = numCM * HPPD['cuidadosMinimos'];
          const horasCI = numCI * HPPD['cuidadosIntermediarios'];
          const horasCAD = numCAD * HPPD['altaDependencia'];
          const horasCSI = numCSI * HPPD['semiIntensivos'];

          totalHorasEnfermagemDia = horasCM + horasCI + horasCAD + horasCSI;

          const cargas = [
            { tipo: 'cuidadosMinimos', horas: horasCM },
            { tipo: 'cuidadosIntermediarios', horas: horasCI },
            { tipo: 'altaDependencia', horas: horasCAD },
            { tipo: 'semiIntensivos', horas: horasCSI }
          ];

          let maiorCarga = -1;
          cargas.forEach(carga => {
            if (carga.horas > maiorCarga) {
              maiorCarga = carga.horas;
              maiorCargaTrabalhoTipo = carga.tipo;
            }
          });
        }

        if (hasError) {
          resultadoDiv.innerText = 'Por favor, corrija os erros nos campos para calcular.';
          resultadoDiv.classList.remove('hidden');
          resultadoDiv.classList.add('bg-danger-red');
          return;
        }

        const horasTrabalhoSemanal = cargaHorariaSemanal[cargaSelecionada];
        const horasTrabalhoDia = horasTrabalhoSemanal / 7;
        const profissionaisTotais = (totalHorasEnfermagemDia / horasTrabalhoDia) * 1.15; // Aplicando IST de 15%

        const percentualEnfermeiro = percentuais[maiorCargaTrabalhoTipo].enfermeiro;
        const percentualTecnico = percentuais[maiorCargaTrabalhoTipo].tecnicoAuxiliar;

        const numeroEnfermeiros = Math.ceil(profissionaisTotais * percentualEnfermeiro);
        const numeroTecnicos = Math.ceil(profissionaisTotais * percentualTecnico);

        // Exibição do resultado
        resultadoDiv.innerHTML = `
                    <p><strong>Total de Horas de Enfermagem Necessárias por Dia:</strong> ${totalHorasEnfermagemDia.toFixed(2).replace('.', ',')} horas</p>
                    <p><strong>Total de Profissionais de Enfermagem Necessários (com IST):</strong> ${Math.ceil(profissionaisTotais)}</p>
                    <p><strong>Número de Enfermeiros Necessários:</strong> ${numeroEnfermeiros}</p>
                    <p><strong>Número de Técnicos/Auxiliares de Enfermagem Necessários:</strong> ${numeroTecnicos}</p>
                `;
        resultadoDiv.classList.remove('hidden', 'bg-danger-red');
        resultadoDiv.classList.add('bg-success-green');

        resultadoDiv.dataset.totalProfissionais = Math.ceil(profissionaisTotais);
        resultadoDiv.dataset.enfermeiros = numeroEnfermeiros;
        resultadoDiv.dataset.tecnicosAux = numeroTecnicos;
        resultadoDiv.dataset.scaleName = "Dimensionamento de Enfermagem";
      }

      function limparDimensionamento() {
        document.getElementById('cargaHoraria').value = '30';
        document.getElementById('tipoUnidade').value = 'unidadeInternacao';

        resultadoDiv.innerHTML = '';
        resultadoDiv.className = 'resultado hidden mt-6';

        document.getElementById('cargaHoraria-error').classList.add('hidden');
        document.getElementById('tipoUnidade-error').classList.add('hidden');
        document.getElementById('numCM-error').classList.add('hidden');
        document.getElementById('numCI-error').classList.add('hidden');
        document.getElementById('numCAD-error').classList.add('hidden');
        document.getElementById('numCSI-error').classList.add('hidden');
        document.getElementById('numLeitosUTI-error').classList.add('hidden');

        delete resultadoDiv.dataset.totalProfissionais;
        delete resultadoDiv.dataset.enfermeiros;
        delete resultadoDiv.dataset.tecnicos;
        delete resultadoDiv.dataset.scaleName;
      }

  <script>
    document.addEventListener("DOMContentLoaded", () => {
      const resultadoDiv = document.getElementById("resultado");
      const btnCalcular = document.getElementById("btnCalcular");
      const btnLimpar = document.getElementById("btnLimpar");
      const btnGerarPDF = document.getElementById("btnGerarPDF");
      const btnNandaNicNoc = document.getElementById("btnNandaNicNoc");
      const globalError = document.getElementById("global-error");

      /* * BLOQUEO DE SCROLL EN INPUTS NUMÉRICOS
       * Evita alteración accidental de valores al desplazar la página.
       */
      document.querySelectorAll('input[type=number]').forEach(input => {
        input.addEventListener('wheel', function(e) {
          // Impide el comportamiento estándar
          e.preventDefault();
          // Remueve el foco
          this.blur();
        });
      });

      // Mapeamento de todos os inputs
      const inputs = {
        ph: document.getElementById("ph"),
        paco2: document.getElementById("paco2"),
        po2: document.getElementById("po2"),
        hco3: document.getElementById("hco3"),
        be: document.getElementById("be"),
        ctco2: document.getElementById("ctco2"),
        sato2: document.getElementById("sato2"),
        na: document.getElementById("na"),
        k: document.getElementById("k"),
        ca: document.getElementById("ca"),
        cl: document.getElementById("cl"),
        glu: document.getElementById("glu"),
        lactato: document.getElementById("lactato"),
        thb: document.getElementById("thb"),
        fo2hb: document.getElementById("fo2hb"),
        fcohb: document.getElementById("fcohb"),
        fmethb: document.getElementById("fmethb"),
        fhhb: document.getElementById("fhhb"),
      };

      function interpretarGasometria() {
        const valores = {};

        // Coleta de valores
        for (const key in inputs) {
          const val = parseFloat(inputs[key].value);
          valores[key] = isNaN(val) ? null : val;
        }

        // Validação mínima
        if (valores.ph === null || valores.paco2 === null || valores.hco3 === null) {
            globalError.classList.remove("hidden");
            resultadoDiv.style.display = "none";
            return;
        } else {
            globalError.classList.add("hidden");
        }

        let diagnosticos = [];
        let condutas = [];
        let detalhes = [];

        // --- PASSO A: Avaliar o pH ---
        let statusPH = "Normal";
        if (valores.ph < 7.35) statusPH = "Acidosis";
        else if (valores.ph > 7.45) statusPH = "Alcalosis";

        // --- PASSO B: Distúrbio Primário & C: Compensação ---
        let disturbio = "";

        // Acidose (Acidosis)
        if (statusPH === "Acidosis") {
            const acidoseResp = valores.paco2 > 45; // pCO2 alto
            const acidoseMet = valores.hco3 < 22 || (valores.be !== null && valores.be < -3); // HCO3 baixo

            if (acidoseResp && !acidoseMet) {
                // Respiratória Pura ou Compensando
                if (valores.hco3 > 26) disturbio = "Acidosis Respiratoria Parcialmente Compensada";
                else disturbio = "Acidosis Respiratoria Descompensada (Aguda)";
                condutas.push("Evaluar ventilación (VNI/IOT), tratar causa pulmonar.");
            } else if (acidoseMet && !acidoseResp) {
                // Metabólica Pura ou Compensando
                if (valores.paco2 < 35) disturbio = "Acidosis Metabólica Parcialmente Compensada";
                else disturbio = "Acidosis Metabólica Descompensada";
                condutas.push("Investigar causa (lactato, cetosis, renal), considerar reposición de base si grave.");
            } else if (acidoseResp && acidoseMet) {
                disturbio = "Acidosis Mixta (Respiratoria + Metabólica)";
                condutas.push("Emergencia: Soporte ventilatorio y corrección metabólica agresiva.");
            } else {
                disturbio = "Acidosis (Trastorno no clasificado)";
            }
        }
        // Alcalose (Alcalosis)
        else if (statusPH === "Alcalosis") {
            const alcaloseResp = valores.paco2 < 35; // pCO2 bajo
            const alcaloseMet = valores.hco3 > 26 || (valores.be !== null && valores.be > 3); // HCO3 alto

            if (alcaloseResp && !alcaloseMet) {
                if (valores.hco3 < 22) disturbio = "Alcalosis Respiratoria Parcialmente Compensada";
                else disturbio = "Alcalosis Respiratoria Descompensada";
                condutas.push("Evaluar ansiedad, dolor, ajuste ventilatorio.");
            } else if (alcaloseMet && !alcaloseResp) {
                if (valores.paco2 > 45) disturbio = "Alcalosis Metabólica Parcialmente Compensada";
                else disturbio = "Alcalosis Metabólica Descompensada";
                condutas.push("Corregir K+, Cl-, volumen, suspender diuréticos si indicado.");
            } else if (alcaloseResp && alcaloseMet) {
                disturbio = "Alcalosis Mixta";
                condutas.push("Tratar causas base simultáneas.");
            } else {
                disturbio = "Alcalosis (Trastorno no clasificado)";
            }
        }
        // pH Normal
        else {
            if (valores.paco2 > 45 && valores.hco3 > 26) disturbio = "Acidosis Respiratoria Compensada (pH normalizado)";
            else if (valores.paco2 < 35 && valores.hco3 < 22) disturbio = "Alcalosis Respiratoria Compensada (pH normalizado)";
            else if (valores.paco2 >= 35 && valores.paco2 <= 45 && valores.hco3 >= 22 && valores.hco3 <= 26) disturbio = "Equilibrio Ácido-Base Normal";
            else disturbio = "Trastorno Mixto compensado o enmascarado";
        }

        diagnosticos.push(disturbio);

        // --- PASSO D: Oxigenação ---
        if (valores.po2 !== null) {
            if (valores.po2 < 80) {
                diagnosticos.push("Hipoxemia");
                condutas.push("Suplementar O₂.");
            }
            if (valores.po2 > 100) {
                detalhes.push("Hiperoxemia (O₂ elevado).");
            }
        }

        if (valores.sato2 !== null && valores.sato2 < 95) {
            detalhes.push("Saturación de O₂ Baja.");
        }

        if (valores.thb !== null) {
             if (valores.thb < 12) detalhes.push("Anemia posible (tHb bajo).");
             if (valores.thb > 17.5) detalhes.push("Policitemia posible (tHb alto).");
        }

        // --- PASSO E: Co-oximetria ---
        if (valores.fcohb !== null && valores.fcohb > 1.5) {
            diagnosticos.push("Elevación de Carboxihemoglobina (Intoxicación CO/Tabaquismo?)");
        }
        if (valores.fmethb !== null && valores.fmethb > 1.5) {
            diagnosticos.push("Metahemoglobinemia");
        }

        // --- PASSO F: Eletrólitos e Lactato ---
        // Lactato
        if (valores.lactato !== null && valores.lactato > 2.0) {
            diagnosticos.push("Hiperlactatemia (Hipoxia tisular/Sepsis?)");
        }

        // Sódio
        if (valores.na !== null) {
            if (valores.na < 135) detalhes.push("Hiponatremia");
            if (valores.na > 145) detalhes.push("Hipernatremia");
        }
        // Potássio
        if (valores.k !== null) {
            if (valores.k < 3.5) detalhes.push("Hipopotasemia");
            if (valores.k > 5.0) detalhes.push("Hiperpotasemia");
        }
        // Cálcio
        if (valores.ca !== null) {
            if (valores.ca < 1.1) detalhes.push("Hipocalcemia");
            if (valores.ca > 1.35) detalhes.push("Hipercalcemia");
        }
        // Cloro
        if (valores.cl !== null) {
            if (valores.cl < 96) detalhes.push("Hipocloremia");
            if (valores.cl > 106) detalhes.push("Hipercloremia");
        }
        // Glicose
        if (valores.glu !== null) {
            if (valores.glu < 70) detalhes.push("Hipoglucemia");
            if (valores.glu > 110) detalhes.push("Hiperglucemia");
        }

        // Formatação Final
        if (condutas.length === 0) condutas.push("Mantener monitoreo.");

        // HTML Compacto Estilo Gotejamento
        let htmlResultado = `
            <p>
                <span class="res-titulo">Diagnóstico:</span> <span class="res-conteudo">${diagnosticos.join(", ")}</span><br>
                ${detalhes.length > 0 ? `<span class="res-titulo">Hallazgos:</span> <span class="res-conteudo">${detalhes.join(", ")}</span><br>` : ''}
                <span class="res-titulo">Conducta:</span> <span class="res-conteudo">${condutas.join(". ")}</span>
            </p>
          `;

        resultadoDiv.innerHTML = htmlResultado;
        resultadoDiv.style.display = "block";

        resultadoDiv.dataset.resultado = disturbio;
      }

      function limparCampos() {
        for (const key in inputs) {
          inputs[key].value = "";
        }
        globalError.classList.add("hidden");
        resultadoDiv.innerHTML = "";
        resultadoDiv.style.display = "none";
        delete resultadoDiv.dataset.resultado;
      }

      btnCalcular?.addEventListener("click", interpretarGasometria);
      btnLimpar?.addEventListener("click", limparCampos);

      btnGerarPDF?.addEventListener("click", () => {
        if (!resultadoDiv.dataset.resultado) {
          alert('Calcule primero para generar PDF.');
          return;
        }
        if (typeof gerarPDFGlobal === "function") {
          gerarPDFGlobal({
            titulo: "Informe de Gasometría Completa",
            subtitulo: "Análisis Ácido-Base, Oxigenación y Electrolitos",
            nomeArquivo: "gasometria_completa.pdf",
            seletorConteudo: ".gasometria-card", // Captura el card llenado
            conteudoExtra: resultadoDiv.innerHTML
          });
        }
      });

      btnNandaNicNoc?.addEventListener("click", () => {
        const resultado = resultadoDiv.dataset.resultado;
        if (!resultado) {
          alert('Calcule primero.');
          return;
        }
        const searchQuery = encodeURIComponent(
          `diagnóstico de enfermería relacionado a ${resultado} NANDA NIC NOC`,
        );
        window.open(
          `https://www.google.com/search?q=${searchQuery}`,
          "_blank",
        );
      });
    });
  </script>
