      // Inicialização ao carregar a página
      document.addEventListener("DOMContentLoaded", () => {

        const searchInput = document.getElementById("searchInput");
        searchInput.addEventListener("input", handleSearch);

      // Função de Pesquisa Avançada
      function handleSearch(e) {
        const term = e.target.value.toLowerCase().trim();
        const summary = document.getElementById("searchSummary");

        if (term.length < 2) {
          renderTaxonomyTree(dbNANDA);
          summary.classList.add("hidden");
          return;
        }

        let matchCount = 0;
        const filteredData = [];

        dbNANDA.forEach((domain) => {
          const matchedClasses = [];

          domain.classes.forEach((cls) => {
            const matchedDiags = cls.diagnoses.filter((diag) => {
              // Lógica de busca: Procura no Título, Definição ou em qualquer Fator/Sintoma
              const inTitle = diag.title.toLowerCase().includes(term);
              const inDef = diag.definition.toLowerCase().includes(term);
              const inChars = diag.characteristics && diag.characteristics.some((c) => c.toLowerCase().includes(term));
              const inFactors = diag.factors && diag.factors.some((f) => f.toLowerCase().includes(term));
              const inPop = diag.populations && diag.populations.some((p) => p.toLowerCase().includes(term));

              return inTitle || inDef || inChars || inFactors || inPop;
            });

            if (matchedDiags.length > 0) {
              matchCount += matchedDiags.length;
              matchedClasses.push({ ...cls, diagnoses: matchedDiags });
            }
          });

          if (matchedClasses.length > 0) {
            filteredData.push({ ...domain, classes: matchedClasses });
          }
        });

        renderTaxonomyTree(filteredData, true);

        summary.innerHTML = `<i class="fa-solid fa-filter text-blue-500 mr-1"></i> Encontrados <b>${matchCount}</b> diagnósticos relacionados a "${term}".`;
        summary.classList.remove("hidden");
      }

      // Renderizar a Árvore
      function renderTreeHTML(data, autoOpen = false) {
        let html = "";
        data.forEach((domain) => {
          html += `
          <div class="tax-domain">
            <div class="tax-domain-header" onclick="toggleDisplay(this.nextElementSibling)">
              <span><i class="fa-solid fa-folder-tree text-blue-400 mr-2"></i> ${domain.domain}</span>
              <i class="fa-solid fa-chevron-down text-slate-400 text-sm"></i>
            </div>
            <div class="tax-classes" style="display: ${autoOpen ? "block" : "none"}">
        `;

          domain.classes.forEach((cls) => {
            html += `
            <div class="tax-class">
              <div class="tax-class-header" onclick="toggleDisplay(this.nextElementSibling)">
                <span><i class="fa-solid fa-layer-group text-slate-300 mr-2"></i> ${cls.name}</span>
                <i class="fa-solid fa-plus text-slate-400 text-xs"></i>
              </div>
              <div class="tax-diagnoses" style="display: ${autoOpen ? "block" : "none"}">
          `;

            cls.diagnoses.forEach((diag) => {
              const isSel = selectedDiagCode === diag.code ? "active" : "";
              html += `
                <div class="tax-diag-item ${isSel}" onclick="selectDiagnosis('${diag.code}')">
                  <i class="fa-solid fa-stethoscope"></i> ${diag.title}
                </div>
            `;
            });

            html += `</div></div>`;
          });
          html += `</div></div>`;
        });
        return html;
      }

      function renderTaxonomyTree(data, autoOpen = false) {
        const container = document.getElementById("taxTreeContainer");
        if (data.length === 0) {
          container.innerHTML = `<div class="p-4 text-center text-sm font-bold text-slate-400 bg-white rounded-xl border border-slate-200">Nenhum diagnóstico compatível com a pesquisa.</div>`;
          return;
        }
        container.innerHTML = renderTreeHTML(data, autoOpen);
      }

      function toggleDisplay(element) {
        if (element.style.display === "block") {
          element.style.display = "none";
        } else {
          element.style.display = "block";
        }
      }

      function findDiagByCode(code) {
        for (let d of dbNANDA) {
          for (let c of d.classes) {
            const found = c.diagnoses.find((x) => x.code === code);
            if (found) return { diag: found, domain: d.domain, class: c.name };
          }
        }
        return null;
      }

      window.selectDiagnosis = function (code) {
        selectedDiagCode = code;

        // Atualizar classes "active" na arvore
        document.querySelectorAll(".tax-diag-item").forEach((el) => el.classList.remove("active"));
        const clickedEl = event.currentTarget;
        if (clickedEl) clickedEl.classList.add("active");

        const data = findDiagByCode(code);
        if (!data) return;

        const d = data.diag;

        // Ocultar Empty State e mostrar Conteúdo
        document.getElementById("emptyState").classList.add("hidden");
        document.getElementById("diagContent").classList.remove("hidden");

        // Preencher Textos
        document.getElementById("diagCode").innerText = "Cód: " + d.code;
        document.getElementById("diagTitle").innerText = d.title;
        document.getElementById("diagDom").innerText = data.domain.split(".")[1].trim(); // Extrai nome do dominio
        document.getElementById("diagClass").innerText = data.class.split(".")[1].trim(); // Extrai nome da classe
        document.getElementById("diagDef").innerHTML = `<b>Definição NANDA:</b> ${d.definition}`;

        // Listas
        document.getElementById("titleChars").innerText = d.code.startsWith("Risco") ? "Fatores de Risco" : "Características Definidoras (Sintomas)";
        document.getElementById("listChars").innerHTML = d.characteristics.map((c) => `<div class="list-badge"><i class="fa-solid fa-angle-right"></i> ${c}</div>`).join("");

        document.getElementById("listFactors").innerHTML = d.factors.map((f) => `<div class="list-badge"><i class="fa-solid fa-link"></i> ${f}</div>`).join("");

        if (d.populations && d.populations.length > 0) {
          document.getElementById("popRiskContainer").classList.remove("hidden");

          document.getElementById("popRiskContainer").classList.add("hidden");
        }

        // NIC / NOC
        document.getElementById("listNOC").innerHTML = d.noc.map((n) => `<li>${n}</li>`).join("");
        document.getElementById("listNIC").innerHTML = d.nic.map((n) => `<li>${n}</li>`).join("");

        // Fab Mobile
        if (window.innerWidth <= 1024) {
          document.getElementById("fabMobile").style.display = "block";
        }

        renderRadarChart(d);

        // Scroll p/ topo no mobile
        if (window.innerWidth <= 1024) {
          document.getElementById("detailsPanel").scrollIntoView({ behavior: "smooth", block: "start" });
        }
      };

        const numNic = diag.nic.length;
        const numNoc = diag.noc.length;

        chartInstance = new Chart(ctx, {

              },
            },
            plugins: {
              legend: { display: false },

          },
        });
